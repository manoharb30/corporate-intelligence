  <action>
    One-time Cypher write: set `methodology_version = 'v1.1'` on every mature strong_buy row that doesn't already have it.

    ```
    cd backend && source venv/bin/activate
    python -c "
    import asyncio
    from app.db.neo4j_client import Neo4jClient

    async def main():
        await Neo4jClient.connect()

        # Count before
        before = await Neo4jClient.execute_query('''
            MATCH (sp:SignalPerformance)
            WHERE sp.direction='buy' AND sp.conviction_tier='strong_buy' AND sp.is_mature=true
            RETURN count(sp) AS total,
                   count(sp.methodology_version) AS with_version
        ''')
        print(f'Before: {before[0][\"total\"]} mature strong_buy, {before[0][\"with_version\"]} with methodology_version')

        # Additive write
        r = await Neo4jClient.execute_query('''
            MATCH (sp:SignalPerformance)
            WHERE sp.direction='buy' AND sp.conviction_tier='strong_buy' AND sp.is_mature=true
              AND sp.methodology_version IS NULL
            SET sp.methodology_version = 'v1.1'
            RETURN count(sp) AS set_count
        ''')
        print(f'Set methodology_version on {r[0][\"set_count\"]} rows')

        # Verify
        after = await Neo4jClient.execute_query('''
            MATCH (sp:SignalPerformance)
            WHERE sp.direction='buy' AND sp.conviction_tier='strong_buy' AND sp.is_mature=true
            RETURN count(sp) AS total,
                   count(sp.methodology_version) AS with_version
        ''')
        print(f'After: {after[0][\"total\"]} mature strong_buy, {after[0][\"with_version\"]} with methodology_version')
        assert after[0]['with_version'] == after[0]['total'], 'coverage must be 100%'
        print('PASS: every mature strong_buy has methodology_version')

        await Neo4jClient.disconnect()

    asyncio.run(main())
    "
    ```

    - Applying to immature rows (let them pick up the value on next compute_all write).
